"""Sector money flow page — aggregation and HTML rendering.

Combines:
- naver_supply_cache.json: per-stock 5-day foreign/institution net value
- krx_sector_cache.json: code → KRX standard sector
- calendar events (from calendar_page.build_calendar_events): future catalysts

Output: public/news_sector_flow.html
"""
import json
from datetime import date, datetime
from pathlib import Path
from typing import Iterable


# Intensity thresholds in percentage of (sector net value / (stock count × 100억))
# Boundary semantics: strong_buy >= +5, buy >= +1, neutral > -1, sell <= -1, strong_sell <= -5
INTENSITY_LABELS = {
    "strong_buy":  ("🔥 강매수", "strong_buy"),
    "buy":         ("📈 매수",   "buy"),
    "neutral":     ("⚪ 중립",    "neutral"),
    "sell":        ("📉 매도",   "sell"),
    "strong_sell": ("❄ 강매도",  "strong_sell"),
}


def classify_intensity(intensity_pct: float) -> tuple[str, str]:
    """Return (display_label, level_id) for a 5-stage intensity classification.

    Boundary rules:
      intensity >= +5  → strong_buy
      +1 <= intensity < +5 → buy
      -1 < intensity < +1  → neutral
      -5 < intensity <= -1 → sell
      intensity <= -5 → strong_sell
    """
    if intensity_pct >= 5.0:
        return INTENSITY_LABELS["strong_buy"]
    if intensity_pct >= 1.0:
        return INTENSITY_LABELS["buy"]
    if intensity_pct > -1.0:
        return INTENSITY_LABELS["neutral"]
    if intensity_pct > -5.0:
        return INTENSITY_LABELS["sell"]
    return INTENSITY_LABELS["strong_sell"]
```

def aggregate_sector_flows(
    supply_cache: dict,
    sector_map: dict[str, dict],
    name_map: dict[str, str],
    window: int,
    as_of: str,
) -> list[dict]:
    """Group naver supply entries by KRX sector and produce per-sector aggregates.

    Returns list of dicts sorted by combined_net_value descending. Each dict:
      {sector, stock_count, foreign_net_value, institution_net_value,
       combined_net_value, intensity_pct, intensity_label, intensity_level,
       top_buy: [{code, name, net_value}, ...3], top_sell: [...3],
       sparkline: [int, ...] }
    """
    # Step 1: filter cache entries for as_of + window
    suffix_window = f"|{window}"
    suffix_date = f"|{as_of}"
    relevant: list[tuple[str, dict]] = []
    for key, entry in supply_cache.items():
        if not isinstance(key, str) or "|" not in key:
            continue
        if not key.endswith(suffix_window):
            continue
        if suffix_date not in key:
            continue
        code = key.split("|", 1)[0]
        relevant.append((code, entry))

    # Step 2: bucket by sector
    by_sector: dict[str, list[dict]] = {}
    for code, entry in relevant:
        sector = (sector_map.get(code) or {}).get("sector") or "기타"
        f_value = int(entry.get("foreign_net_value") or 0)
        i_value = int(entry.get("institution_net_value") or 0)
        combined = f_value + i_value
        by_sector.setdefault(sector, []).append({
            "code": code,
            "name": name_map.get(code, code),
            "foreign_net_value": f_value,
            "institution_net_value": i_value,
            "net_value": combined,
            "rows": entry.get("rows") or [],
        })

    # Step 3: compute per-sector aggregates
    results: list[dict] = []
    for sector, stocks in by_sector.items():
        f_total = sum(s["foreign_net_value"] for s in stocks)
        i_total = sum(s["institution_net_value"] for s in stocks)
        combined = f_total + i_total
        count = len(stocks)
        # intensity = combined_net_value / (count × 100억) × 100
        intensity_pct = (combined / max(1, count * 10_000_000_000)) * 100
        label, level = classify_intensity(intensity_pct)

        sorted_stocks = sorted(stocks, key=lambda s: s["net_value"], reverse=True)
        top_buy = [
            {"code": s["code"], "name": s["name"], "net_value": s["net_value"]}
            for s in sorted_stocks[:3] if s["net_value"] > 0
        ]
        top_sell = [
            {"code": s["code"], "name": s["name"], "net_value": s["net_value"]}
            for s in reversed(sorted_stocks[-3:]) if s["net_value"] < 0
        ]

        # Sparkline: aggregate per-day combined net value across stocks
        # rows[i] = {date, close, foreign_net (shares), institution_net (shares)}
        sparkline = _build_sparkline(stocks)

        results.append({
            "sector": sector,
            "stock_count": count,
            "foreign_net_value": f_total,
            "institution_net_value": i_total,
            "combined_net_value": combined,
            "intensity_pct": round(intensity_pct, 2),
            "intensity_label": label,
            "intensity_level": level,
            "top_buy": top_buy,
            "top_sell": top_sell,
            "sparkline": sparkline,
        })

    results.sort(key=lambda r: r["combined_net_value"], reverse=True)
    return results


def _build_sparkline(stocks: list[dict]) -> list[int]:
    """Sum per-day combined net value (foreign + institution) × close across stocks.

    Returns a list of daily totals (oldest → newest, length = window). Empty if no rows.
    """
    by_date: dict[str, int] = {}
    for s in stocks:
        for row in s.get("rows", []):
            d = row.get("date")
            close = row.get("close") or 0
            f_net = row.get("foreign_net") or 0
            i_net = row.get("institution_net") or 0
            day_value = (f_net + i_net) * close
            if d:
                by_date[d] = by_date.get(d, 0) + int(day_value)
    return [by_date[d] for d in sorted(by_date.keys())]

def group_events_by_sector(
    events: list[dict],
    sector_map: dict[str, dict],
    today: date | None = None,
    days_ahead: int = 14,
) -> dict[str, list[dict]]:
    """Map calendar events to sectors. Returns {sector: [event, ...]}.

    Mapping rules (V1, conservative):
      1. If event has direct stocks → use those stocks' sectors
      2. Else if event has inferred_stocks → use those
      3. Else if event has sector_hints (list of sector names) → use directly
      4. Else → drop (do not assign to any sector)
    Filter: event_date must be within [today, today + days_ahead] inclusive.
    """
    from datetime import date as _date
    today = today or _date.today()
    cutoff = today + (date.fromisoformat(today.isoformat()).resolution * 0)  # placeholder
    # Use simple timedelta:
    from datetime import timedelta as _td
    cutoff_date = today + _td(days=days_ahead)

    grouped: dict[str, list[dict]] = {}

    for ev in events:
        ed_raw = ev.get("event_date")
        if not ed_raw:
            continue
        try:
            ed = _date.fromisoformat(ed_raw[:10])
        except (ValueError, TypeError):
            continue
        if ed < today or ed > cutoff_date:
            continue

        sectors: set[str] = set()

        for key in ("stocks", "inferred_stocks"):
            for s in (ev.get(key) or []):
                code = s.get("code") if isinstance(s, dict) else None
                if not code:
                    continue
                sec = (sector_map.get(code) or {}).get("sector")
                if sec:
                    sectors.add(sec)
            if sectors:
                break  # prefer direct over inferred

        if not sectors:
            for hint in (ev.get("sector_hints") or []):
                if hint:
                    sectors.add(hint)

        for sec in sectors:
            grouped.setdefault(sec, []).append(ev)

    # Sort each sector's events by date asc
    for sec in grouped:
        grouped[sec].sort(key=lambda e: e.get("event_date") or "")

    return grouped
```

def _sparkline_svg(points: list[int], width: int = 120, height: int = 24) -> str:
    if not points:

def build_sector_flow_page(
    out_path: Path | str,
    supply_cache_path: Path | str = "app/naver_supply_cache.json",
    sector_cache_path: Path | str = "app/krx_sector_cache.json",
    name_map: dict[str, str] | None = None,

    supply = json.loads(Path(supply_cache_path).read_text())
    sector_raw = json.loads(Path(sector_cache_path).read_text()) if Path(sector_cache_path).exists() else {}
    sector_map = {k: v for k, v in sector_raw.items() if not k.startswith("_")}
    name_map = name_map or {}
    as_of = as_of or _latest_as_of(supply)

    flows_5 = aggregate_sector_flows(supply, sector_map, name_map, window=5, as_of=as_of)

        events_by_sector=events_by_sector,
        as_of=as_of,
    )
    Path(out_path).write_text(html, encoding="utf-8")
    return html


def _latest_as_of(supply_cache: dict) -> str:
    dates = set()
    for k in supply_cache.keys():
        if isinstance(k, str) and k.count("|") == 2:
            dates.add(k.split("|")[1])
    return max(dates) if dates else date.today().isoformat()

      - uses: actions/checkout@v4

        with:

      - name: Install dependencies
        run: pip install -r requirements.txt

            echo "No cache changes."
          else
